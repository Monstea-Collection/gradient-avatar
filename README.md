```bash
npm install --save gradient-avatar
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'gradient-avatar'
import 'gradient-avatar/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
